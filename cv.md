function stringBreakers(n, string) {
  const arrStrBreak = [];
  let str = string.replace(/\s+/g, '');
  
  while (str) {
    if (str.length < n) {
        arrStrBreak.push(str);
        break;
    } else {
        arrStrBreak.push(str.substr(0, n));
        str = str.substring(n);
    }
}
  return arrStrBreak.join('\n');
}
